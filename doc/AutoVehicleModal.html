def populate_with(data)
  return super unless data.key? 'vehicles'

  begin
    data['vehicles'].each_with_index do |vehicle, index|
      populate_with(vehicle)

      save_and_add_another unless index == data['vehicles'].count - 1
    end
  rescue Exception => ex
    parent_container.raise_page_errors 'Populating auto driver modal'
    raise ex
  end
end

def territory=(new_territory)
  territory_element.scroll
  territory_element.click!
  Watir::Wait.until { territory_element.include? new_territory }
  territory_element.select_value new_territory
  Watir::Wait.until { territory == new_territory }
end
