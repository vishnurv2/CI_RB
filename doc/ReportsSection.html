def order_reports_by_type(reports, clear_first = true)
  clear_order_statuses if clear_first
  reports = [reports] if reports.is_a?(String)
  reports.each do |report_type|
    find_report_by_type(report_type).order = true
  end
  order_reports
  Watir::Wait.until { !parent_container.div(id: 'divModalContent').nil? && parent_container.div(id: 'divModalContent').present? }
end
