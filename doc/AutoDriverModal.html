def accidents_and_violations=(desired_as_and_vs)
  tabs.active_tab = 'Accidents / Violations'
  wait_for_accident_element
  desired_as_and_vs.each do |ith_accident_or_violation|
    ith_accident_or_violation['accident_or_violation'] == 'accident' ? add_accident : add_violation
    wait_for_accs_and_violations
    accidents_and_violations.last.populate_details(ith_accident_or_violation)
  end
end

def add_accident(type = nil, date = Time.now.strftime('%m/%d/%Y'))
  occurrences = accidents_and_violations.count
  add_accident_button
  Watir::Wait.until { accidents_and_violations.count > occurrences }
  av = accidents_and_violations.last
  return av if type.nil?

  av.occurrence_date = date
  av.occurrence_type = type
  save_and_close
end

def date_of_birth=(date)
  date_of_birth_text_field_element.set(Chronic.parse(date).strftime('%m/%d/%Y'))
  d = Chronic.parse(date)
  first_license_date_text_field_element.set(Chronic.parse("#{d.month}/#{d.day}/#{d.year + 16}").strftime('%m/%d/%Y'))
end

def date_of_birth_matches?(date)

def existing_parties_include?(name_parts)

def populate_with(data)
  middle_name_element.focus  # This is a terrible hack to get around an issue in the application.  As of 8/6/2019 a validation message
                             # appears as soon as the modal is opened, it disappears once another control has focus.
                             # without this line, the modal will rerender just as we try to click a button, causing us to miss the button.

  return super unless data.key? 'drivers'

  data['drivers'].each_with_index do |driver, index|
    populate_with(driver)
    dob = date_of_birth
    self.first_license_date = "#{dob.month}/#{dob.day}/#{dob.year + 16}"
    save_and_add_another_driver unless index == data['drivers'].count - 1
  end
end
