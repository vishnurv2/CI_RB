def _wait_for_pending_ajax(timeout)
  end_time = ::Time.now + timeout
  until ::Time.now > end_time
    begin
      return true if browser.execute_script(::EDSL::PageObject::JavascriptFrameworkFacade.pending_requests).zero?
    rescue Selenium::WebDriver::Error::UnknownError, Selenium::WebDriver::Error::JavascriptError
    end
    sleep 0.5
  end
  false
end

def wait_for_ajax(timeout = Nenv.wait_for_ajax_time, message = nil)
  return if _wait_for_pending_ajax(timeout)

  raise "Full page exception encountered while waiting for AJAX\r\nAccount ID: #{BasePage.account_number}\r\nPolicy Activity ID: #{BasePage.policy_number}" if browser.url.include?('ErrorRecovery')

  raise_page_errors "Waiting for AJAX in #{self.class}"
  message ||= 'Timed out waiting for ajax requests to complete'
  raise message
end
