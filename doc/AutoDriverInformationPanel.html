def find_driver_by_hash(to_find)
  return nil unless drivers_grid.show_entries?

  fields = %w[name date_of_birth]
  to_find['date_of_birth'] = _to_date_with_age(to_find['date_of_birth'])
  to_find['name'] = to_find.slice('name_prefix', 'first_name', 'middle_name', 'last_name', 'name_suffix').values.join(' ')
  drivers_grid.reverse_each_page do |items|
    found_item = items.find { |item| fields.all? { |field| to_find[field] == item.send(field) } }
    return found_item unless found_item.nil?
  end
end

def remove_extra_drivers(number_to_keep = 1)
  return unless drivers_grid?

  drivers = _find_drivers_to_remove
  return if drivers.count <= number_to_keep

  puts "Removing #{drivers.count - number_to_keep} drivers"
  api = PolicyAPI::DELETEApi.new
  drivers[number_to_keep..-1].each do |driver|
    puts "Removing #{driver[:driver_id]}"
    api.policies_delete_driver driver[:act_id], driver[:driver_id]
  end
end
